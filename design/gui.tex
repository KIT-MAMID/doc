The \refgo{gui} is built with AngularJS.
The model is defined within AngularJS by the returned JSON object structure of the master API (see \ref{masterapi:apiobjects}).

"[In AngularJS] the purpose of controllers is to expose variables and functionality to
expressions and directives"\footnote{\url{https://docs.angularjs.org/guide/concepts\#controller}}.

The view functions are abstracted into AngularJS, so the \codeinline{render}-calls in the UML are placeholders for the work of Angluar.
The views themselves are HTML files.

\jsview{Main}{Displays the main page with the latest 5 current slave problems and the latest 5 Replica Set problems as well as
	a pie chart showing the distribution of slave states in the system.}

\jsview{ReplicaSets}{Displays all configured Replica Sets.}

\jsview{ReplicaSet}{Displays a specific Replica Set and allows CRUD of a Replica Set.}

\jsview{Slaves}{Displays all Slaves managed by \mamid.}

\jsview{Slave}{Displays a specific Slave and allows CRUD of a Slave.}

\jsview{Problems}{Displays all problems in the system.}

\jsview{RiskGroups}{Displays all risk groups and allows CRUD.}

\jscontroller{main}{
	\begin{itemize}
		\item Uses \refgoalt{masterapi.slaves.getAll}{[GET] /slaves} to get all slaves.
	\end{itemize}
}

\jscontroller{slaveIndex}{
	\begin{itemize}
		\item Uses \refgoalt{masterapi.slaves.getAll}{[GET] /slaves} to get all slaves.
	\end{itemize}
}

\jscontroller{replicaSetIndex}{
		\begin{itemize}
			\item Uses \refgoalt{masterapi.replicasets.getAll}{[GET] /replicasets} to get all Replica Sets.
		\end{itemize}
}

\jscontroller{problemIndex}{
	\begin{itemize}
		\item Uses \refgoalt{masterapi.problems.getAll}{[GET] /problems} to get all problems.
	\end{itemize}
}

\jscontroller{slaveById}{
		\begin{itemize}
			\item Uses \refgoalt{masterapi.slaves.add}{[PUT] /slaves} to add a new slave to the system.
			\item Uses \refgoalt{masterapi.slaves.getById}{[GET] /slaves/<id>} to get a specific slave.
			\item Uses \refgoalt{masterapi.slaves.update}{[POST] /slaves/<id>} to update a specific slave.
			\item Uses \refgoalt{masterapi.slaves.delete}{[DELETE] /slaves/<id>} to delete a specific slave.
			\item Uses \refgoalt{masterapi.slaves.getProblems}{[GET] /slaves/<id>/problems} to get the problems of a specific slave.
		\end{itemize}
}
\jscontroller{replicaSetById}{
		\begin{itemize}
			\item Uses \refgoalt{masterapi.replicasets.add}{[PUT] /replicasets} to create a new Replica Set.
			\item Uses \refgoalt{masterapi.replicasets.getById}{[GET] /replicasets/<id>} to get a specific Replica Set.
			\item Uses \refgoalt{masterapi.replicasets.update}{[POST] /replicasets/<id>} to update a specific Replica Set.
			\item Uses \refgoalt{masterapi.replicasets.delete}{[DELETE] /replicasets/<id>} to delete a specific Replica Set.
			\item Uses \refgoalt{masterapi.replicasets.getProblems}{[GET] /replicasets/<id>/problems} to get the problems of a specific Replica Set.
		\end{itemize}	
}

\jscontroller{riskGroupIndex}{
	\begin{itemize}
		\item Uses \refgoalt{masterapi.riskgroups.getAll}{[GET] /riskgroups} to get all risk groups.
		\item Uses \refgoalt{masterapi.riskgroups.assignSlave}{[PUT] /riskgroups/<risk\_group\_id>/slave/<slave\_id>} to add a slave to a specific risk group.
		\item Uses \refgoalt{masterapi.riskgroups.removeSlave}{[DELETE] /riskgroups/<risk\_group\_id>/slave/<slave\_id>} to remove a slave from a specific risk group.
		\item Uses \refgoalt{masterapi.riskgroups.add}{[PUT] /riskgroups/<id>} to add a new risk group.
		\item Uses \refgoalt{masterapi.riskgroups.update}{[POST] /riskgroups/<id>} to update a specific risk group.
		\item Uses \refgoalt{masterapi.riskgroups.delete}{[DELETE] /riskgroups/<id>} to remove a specific risk group.
	}
	\end{itemize}
}
