\section{GUI}
\beginpackage{gui}
\begin{figure}[H]
	\includegraphics[width=\textwidth]{gui_classes}
	\caption{GUI}
\end{figure}
The \refgo{gui} is built with AngularJS. The model is defined by the returned JSON object structure of the master API (see \ref{masterapi:apiobjects}).
%TODO remark about AngularJS in the beginning of the package description

\jsview{Main}{Displays the main page with the latest 5 current slave problems and the latest 5 Replica Set problems as well as
	a pie chart showing the distribution of slave states in the system.}

\jsview{ReplicaSets}{Displays all configured Replica Sets.}

\jsview{ReplicaSet}{Displays a specific Replica Set and allows CRUD of a Replica Set.}

\jsview{Slaves}{Displays all Slaves managed by \mamid.}

\jsview{Slave}{Displays a specific Slave and allows CRUD of a Slave.}

\jsview{Problems}{Displays all problems in the system.}

\jsview{RiskGroups}{Displays all risk groups and allows CRUD.}

\jscontroller{main}{
	\begin{itemize}
		\item Uses \refgo{masterapi.slaves.getAll} to get all slaves.
	\end{itemize}
}

\jscontroller{slaveIndex}{
	\begin{itemize}
		\item Uses \refgo{masterapi.slaves.getAll} to get all slaves.
	\end{itemize}
}

\jscontroller{replicaSetIndex}{
		\begin{itemize}
			\item Uses \refgo{masterapi.replicasets.getAll} to get all replica sets.
		\end{itemize}
}

\jscontroller{problemIndex}{
	\begin{itemize}
		\item Uses \refgo{masterapi.problems.getAll} to get all problems.
	\end{itemize}
}

\jscontroller{slaveById}{
		\begin{itemize}
			\item Uses \refgo{masterapi.slaves.add} to add a new slave to the system.
			\item Uses \refgo{masterapi.slaves.getById} to get a specific slave.
			\item Uses \refgo{masterapi.slaves.update} to update a specific slave.
			\item Uses \refgo{masterapi.slaves.delete} to delete a specific slave.
			\item Uses \refgo{masterapi.slaves.getProblems} to get the problems of a specific slave.
		\end{itemize}
}
\jscontroller{replicaSetById}{
		\begin{itemize}
			\item Uses \refgo{masterapi.replicasets.add} to create a new Replica Set.
			\item Uses \refgo{masterapi.replicasets.getById} to get a specific Replica Set.
			\item Uses \refgo{masterapi.replicasets.update} to update a specific Replica Set.
			\item Uses \refgo{masterapi.replicasets.delete} to delete a specific Replica Set.
			\item Uses \refgo{masterapi.replicasets.getProblems} to get the problems of a specific Replica Set.
		\end{itemize}	
}

\jscontroller{riskGroupIndex}{
	\begin{itemize}
		\item Uses \refgo{masterapi.riskgroups.getAll} to get all risk groups.
		\item Uses \refgo{masterapi.riskgroups.assignSlave} to add a slave to a specific risk group.
		\item Uses \refgo{masterapi.riskgroups.removeSlave} to remove a slave from a specific risk group.
		\item Uses \refgo{masterapi.riskgroups.update} to update a specific risk group.
		\item Uses \refgo{masterapi.riskgroups.delete} to remove a specific risk group.
	}
	\end{itemize}
}
