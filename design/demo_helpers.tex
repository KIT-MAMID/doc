\section {Formatting}\label{formatting}
% NOTE Declaring the current package
\renewcommand{\gocurpackage}{formatting}

The remainder of this section is a demonstration of the formatting chosen to describe packages types, methods and parameters.

This text could as well be a description of the package \refgo{formatting}.

\struct{AStruct}{
  Overall description of the \codeinline{struct} \refgo{formatting.AStruct}.
}{
    \property{fieldname}[FieldType]{
      The description of the struct's field \refgo{formatting.AStruct.fieldname}.
    }
}{
    \method{Methodname}[ReturnType](%% NOTE the parentheses ()
        \param{pname1}[ParamType]{Description of the parameter \refparam{pname1}.}
        \param{pname2}[ParamType]{Description of the parameter \refparam{pname2}.}
    ){
        Description of the method \refgo{formatting.AStruct.Methodname}.
        The dot in the parentheses symbolizes that this method expects parameters.
    }
}
