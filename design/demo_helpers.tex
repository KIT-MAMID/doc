\section {Formatting \& Legend}\label{legend}

% NOTE Declaring the current package
\renewcommand{\gocurpackage}{legend}

This is a legend package explaining the formatting of the following packages.

\struct{AStruct}{
    Overall description of a \refgo{legend.AStruct}.
    It implements \refgo{legend.AnInterface}.
}{
    \property{variable}[Type]{The description of the struct's field \emph{variable}}
    \property{fieldtwo}[Type2]{
      The description of the struct's field \refgo{legend.AStruct.fieldtwo}.
    }
}{
    \method{Methodname}[ReturnType]{Method without parameters}
    \method{Methodname2}[ReturnType](%% NOTE the parentheses ()
        \param{pname}[ParamType]{Description of the parameter}
    ){
        Description of the method \refgo{legend.AStruct.Methodname2}.
        The dot in the parentheses symbolizes that this method expects parameters.
    }
}

\interface{AnInterface}{
  Overall description of the Interface. An interface has no fields, hence there is no section for fields.

  References to other types, e.g. \refgo{legend.AStruct} are possible.
  When referencing another package the package name is shown, e.g. \refgo{master.ClusterAllocator}
}{
  \method{Methodname}[ReturnType]{Method without parameters}
  \method{Methodname2}[ReturnType]( %% NOTE the parentheses ()
    \param{pname}[ParamType]{Description of the parameter}
  ){
    Description of the method \refgo{legend.AnInterface.Methodname2}.
  }
}
